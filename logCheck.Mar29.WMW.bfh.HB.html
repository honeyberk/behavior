ttest <- function(coln,group){
  col_C <- coln[group == 'FALSE']
  col_E <- coln[group == 'TRUE']
  t.test(col_E,col_C,alternative = 'greater')$p.value
}

wmwtest <- function(coln, group) {
  col_C <- coln[group == 'FALSE']
  col_E <- coln[group == 'TRUE']
  wilcox.test(col_E, col_C, alternative = 'greater')$p.value
}
